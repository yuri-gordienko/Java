ls - observe files in current directory
ls -la - содержимое каталога со скрытыми файлами
cd - перейти в директории на шаг выше
cd ../package - переход шаг назад в нужную директорю
cd ../../../ - выходить по директориям
pwd - полный путь к файлу (имя класса это его полный путь)

javac HelloWorld.java - компилируем файл файл с расширением .class
java HelloWorld - запускаем файл  с расширением .class (.class уже писать не нужно, подразумевается)

javac ua/com/alevel/HelloWorld.java - запуск компиляции из конкретной дирректории
java ua.com.alevel.HelloWorld - запуск файла

javac -sourcepath ./src -d build/classes ./src/ua/com/alevel/GoodDay.java - компилируем в отдельный пакет с новыми директориями
при этом говорим, что src это корень исходников, через команду sourcepath
java -cp build/classes ua.com.alevel.GoodDay - запускаем программу

MANIFEST.MF - файл для сбора архива .jar
jar cvfm build/jar/good_day2.jar resources/MANIFEST.MF -C build/classes .
сгенерировали .jar, укказали сначала куда сохранить, потом через что сгенерировать, потом откуда взять файл типа .class

java -jar build/jar/good_day.jar - запускаем .jar файл

mvn install - сборка, компиляция и установка проекта в локальний репозиторий
mvn clean - удаляет папку /target, где лежат скомпилированные файлы
mvn compile - компилирует программу
mvn package - собирает .jar
mvn clean package - пересобирает прогу (удаляет /target и создает новые)

git init - активировать гит для доступности из терминала
git commit - зафиксировали изменения локально
git commit -m комит с комментарием
git push -u origin main - пушим в ветку мэйн
git status - посмотреть что в системе сейчас
git add <file> - добавляем конкретный файл для комита

.bash_profile - файл, дзя збереження шляхів глобально, щоб програма була видна з любої директорії, може виконуватись
не автоматично при входженні в систему, коли ви використовуєте Zsh, оскільки вона використовує інший файл налаштувань,
такий як ~/.zshrc.
Zsh (або Z Shell) — це інтерактивна оболонка командного рядка для операційних систем UNIX і подібних до UNIX.

~/.bash_profile - входимо одразу в директорію bash_profile для прописання шляху до файлу .bash_profile
open ~/.bash_profile - відкрити файл .bash_profile, або
nano ~/.bash_profile - одразу відкрити файл .bash_profile
Ctrl + X - зберегти зміни, находячись у файлі .bash_profile

echo 'export PATH=$PATH:/шлях/до/вашої/програми' >> ~/.bash_profile - одразу прописати шлях з терміналу, не заходячи в .bash_profile
source ~/.bash_profile - збереження змін в терміналі, або перезапустити термінал

.zshrc - файл для консолі, файл налаштувань Zsh, щоб консоль бачила програми
nano ~/.zshrc - налаштування автоматичного виклику команд у вашому профілі оболонки
